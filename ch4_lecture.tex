\begin{frame}
\frametitle{Monotone codes}
\begin{itemize}    
% \footnotesize {
% \small{    
    
    \begin{theorem}\label{Elias}
    Let random variable $i$ take values from natural numbers and it's probability distribution satisfies: $p_i \le p_j $ if $i > j$. 
    Then, average length of Elias codewords $\bar {l}$ satisfies 
    \[
    \bar {l} \le H(1 + o(H)),
    \]
    where $H$ is entropy of $i$, and $o(H) \to 0$ when $H \to \infty $.
    \end{theorem}


\end{itemize}
\end{frame}



\begin{frame}
\frametitle{Monotone codes}
Proof.
\begin{itemize}    
% \footnotesize {
\small{    
    
    \item 
    \begin{equation}
    \label{eq4_6} p_i \le \frac{1}{i} \quad , \quad i \le \frac{1}{p_i
    }, \quad i = 1,2,... \quad .
    \end{equation}
    
    \item 
    \begin{eqnarray}
    \label{eq4_7}
    \bar {l} &=& \sum\limits_{i = 1}^\infty {l_i p_i \le }\nonumber\\
    &\le& \sum\limits_{i = 1}^\infty {p_i (\log i + 2\log \log i + 2)}
    \le \nonumber\\
    &\le& \sum\limits_{i = 1}^\infty {p_i \log \frac{1}{p_i } + }
    2\sum\limits_{i = 1}^\infty {p_i \log \left( {1 + \log \frac{1}{p_i
    }} \right) + } 2 \le\nonumber\\
    &\le& H + 2\log (1 + H) + 2.
    \end{eqnarray}
}
\end{itemize}
\end{frame}

    \begin{theorem}Let $H(X)$ be the entropy of one-dimensional distribution of discrete stationary source. Average code rate for interval coding in composition with Elias code satisfies:
    \[
    \bar {R} \le H(X)(1 + o(H(X)),
    \]
    where $o(H) \to 0$ when $H \to \infty $.
    \end{theorem}

\begin{frame}

% \footnotesize {
% \small{    

    \begin{lemma}(Katz Lemma).
    For discrete stationary source such that $p(a) > 0$ holds:
    \begin{equation}
    \label{eq4_15} \bar {r}_a p(a) = {\rm P}(x_n = a\quad \mbox{at least one} \quad n, \quad n = 0,1,...).
    \end{equation}
    for an ergodic source:
    \begin{equation}
    \label{eq4_16} \bar {r}(a) = \frac{1}{p(a)}.
    \end{equation}
    \end{lemma}
